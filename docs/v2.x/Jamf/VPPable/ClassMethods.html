def all_vpp_device_assignable
  data = {}
  all_ids.each do |id|
    obj = fetch id: id
    next unless obj.vpp_device_based?

    data[obj.name] = {
      total: obj.vpp_licenses_total,
      used: obj.vpp_licenses_used,
      remaining: obj.vpp_licenses_remaining
    }
  end
  data
end
