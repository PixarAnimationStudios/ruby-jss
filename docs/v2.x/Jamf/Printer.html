    super

    if self.in_jss?

        @uri = @init_data[:uri]
        @CUPS_name = @init_data[:CUPS_name]

        raise Jamf::InvalidDataError, "os_requirements must be a String." unless @init_data[:os_requirements].is_a?(String) || @init_data[:os_requirements].nil?
