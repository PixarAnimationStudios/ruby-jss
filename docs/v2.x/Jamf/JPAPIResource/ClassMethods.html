def new(**data)
  calling_method = caller_locations(1..1).first.label
  unless NEW_CALLERS.include? calling_method
    raise Jamf::UnsupportedError, 'Use .fetch, .create, or .all(instantiate:true) to instantiate Jamf::JPAPIResource objects'
  end

  super(**data)
end
