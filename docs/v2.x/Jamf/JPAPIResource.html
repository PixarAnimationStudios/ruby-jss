def save
  raise Jamf::UnsupportedError, "#{self.class} objects cannot be changed" unless self.class.mutable?

  if exist?
    return unless unsaved_changes?

    update_in_jamf
  else
    create_in_jamf
  end

  clear_unsaved_changes

  @id || :saved
end
