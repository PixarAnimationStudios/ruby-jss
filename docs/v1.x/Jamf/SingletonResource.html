def self.fetch(reload = false, cnx: Jamf.cnx)
  stop_if_base_class
  cnx.singleton_cache[self] = nil if reload
  cached = cnx.singleton_cache[self]
  return cached if cached

  data = cnx.get rsrc_path
  cnx.singleton_cache[self] = new data, cnx: cnx
end
