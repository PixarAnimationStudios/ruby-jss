def self.scriptContents(scr_ident, cnx: Jamf.cnx)
  id = valid_id scr_ident
  raise Jamf::NoSuchItemError, "No script matches '#{scr_ident}'" unless id
  cnx.download "#{RSRC_VERSION}/#{RSRC_PATH}/#{id}/download"
end
